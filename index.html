-- Création des tables -- 1. Table des formations CREATE TABLE education ( id SERIAL PRIMARY KEY, institution VARCHAR(255) NOT NULL, degree JSONB NOT NULL, -- Exemple : {"fr": "Bachelier...", "en": "Bachelor...", "nl": "..."} description JSONB, start_date DATE, end_date DATE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- 2. Table des expériences professionnelles CREATE TABLE experience ( id SERIAL PRIMARY KEY, company VARCHAR(255) NOT NULL, title JSONB NOT NULL, description JSONB, start_date DATE, end_date DATE ); -- 3. Table des projets CREATE TABLE projects ( id SERIAL PRIMARY KEY, title JSONB NOT NULL, category VARCHAR(100), tech_stack TEXT[], -- Utilisation d'un tableau pour les tags (Python, SQL, etc.) status VARCHAR(50), -- "en cours", "bientôt", "terminé" link_url VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- 4. Table des compétences CREATE TABLE skills ( id SERIAL PRIMARY KEY, name VARCHAR(100) NOT NULL, category VARCHAR(50), -- "technique", "soft_skill", "langue" proficiency_level VARCHAR(50) ); -- 5. Table des certifications CREATE TABLE certifications ( id SERIAL PRIMARY KEY, name VARCHAR(255) NOT NULL, issuer VARCHAR(100), date_obtained DATE ); -- 6. Table des messages (Formulaire de contact) CREATE TABLE contact_messages ( id SERIAL PRIMARY KEY, sender_name VARCHAR(150) NOT NULL, sender_email VARCHAR(255) NOT NULL, subject VARCHAR(255), message TEXT NOT NULL, is_read BOOLEAN DEFAULT FALSE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
